import React, { useState, useMemo, useCallback, useRef } from 'react';
import { useNavigate } from 'react-router-dom';
import { useAuth } from '../../context/AuthContext';
import { useDiscovery } from '../../context/DiscoveryContext';
import { useCategories } from '../../hooks/useCategories';
import { useProducts } from '../../hooks/useProducts';
import { useSoldCounts } from '../../hooks/useSoldCounts';
import { useDebounce } from '../../hooks/useDebounce';
import { useIntersectionReveal } from '../../hooks/useIntersectionReveal';
import { 
  Package, 
  Banknote, 
  TrendingUp, 
  Activity, 
  Printer
} from 'lucide-react';

// Components
import HeroCarousel from '../../components/HeroCarousel/HeroCarousel';
import QuickCategoryRow from '../../components/QuickCategoryRow/QuickCategoryRow';
import ProductGrid from '../../components/VirtualizedProductGrid/ProductGrid';
import Pagination from '../../components/Pagination/Pagination';
import Footer from '../../components/Footer/Footer';
import SearchBar from './SearchBar';
import FilterDropdown from '../../components/FilterDropdown/FilterDropdown';

// Assets
import hero1 from '../../assets/hero/hero-1.jpg';
import hero2 from '../../assets/hero/hero-2.jpg';
import hero3 from '../../assets/hero/hero-3.jpg';

// Types
interface IHomepageFilters {
  category_id: string | null;
  price_min: number | null;
  price_max: number | null;
  sort: string;
  status: string;
  screenplate_id: string | null;
}

// ── Module-level constants ──────────────────────────────────────────────────

const HERO_SLIDES = [
  { 
    src: hero1, 
    title: 'Industrial Print. Zero Compromise.', 
    sub: 'Custom cup printing for brands that demand precision.' 
  },
  { 
    src: hero2, 
    title: 'Every Surface. Every Brand.', 
    sub: 'From milktea cups to packaging — built for volume.' 
  },
  { 
    src: hero3, 
    title: 'Your Logo. Our Craft.', 
    sub: 'Screenplate printing with same-day turnaround.' 
  },
] as const;

const PRICE_RANGES = [
  { label: 'All Prices', min: null, max: null },
  { label: 'Under ₱100', min: 0, max: 99 },
  { label: '₱100 - ₱500', min: 100, max: 500 },
  { label: '₱500 - ₱1,000', min: 500, max: 1000 },
  { label: 'Over ₱1,000', min: 1000, max: null },
] as const;

const SORT_OPTIONS = [
  { label: 'Price: Low to High', value: 'price_asc' },
  { label: 'Price: High to Low', value: 'price_desc' },
  { label: 'Newest Arrivals', value: 'newest' },
  { label: 'Most Popular', value: 'popular' },
] as const;

const STATUS_OPTIONS = [
  { label: 'All Status', value: 'all' },
  { label: 'In Stock', value: 'in_stock' },
  { label: 'Out of Stock', value: 'out_of_stock' },
] as const;

const PLATE_OPTIONS = [
  { label: 'All Plates', value: 'all' },
  { label: 'Requires Plate', value: 'requires' },
  { label: 'Stock Item', value: 'no_plate' },
] as const;

// ── Component ───────────────────────────────────────────────────────────────

const Homepage: React.FC = () => {
  const { user } = useAuth();
  const { openDiscovery } = useDiscovery();
  const navigate = useNavigate();
  const productGridRef = useRef<HTMLDivElement>(null);

  // Auth guard wrapper
  const checkAuthAndRun = useCallback((action: () => void) => {
    if (!user?.isLoggedIn) {
      navigate('/login'); 
      return;
    }
    action();
  }, [user?.isLoggedIn, navigate]);

  // Search state
  const [rawSearch, setRawSearch] = useState('');
  const debouncedSearch = useDebounce(rawSearch, 350);

  // Filters state
  const [filters, setFilters] = useState<IHomepageFilters>({
    category_id: null,
    price_min: null,
    price_max: null,
    sort: 'price_asc',
    status: 'all',
    screenplate_id: null,
  });

  const [currentPage, setCurrentPage] = useState(1);
  const itemsPerPage = window.innerWidth < 768 ? 10 : 20;

  // Data fetching
  const { categories, isLoading: catLoading } = useCategories();
  const { soldMap } = useSoldCounts();
  const { products, totalPages, isLoading: prodLoading } = useProducts({
    search: debouncedSearch.replace(/[^a-zA-Z0-9\s\-_.₱]/g, ''),
    category: filters.category_id || undefined,
    sort: filters.sort,
    status: filters.status,
    screenplate_id: filters.screenplate_id,
    price_min: filters.price_min ?? undefined,
    price_max: filters.price_max ?? undefined,
    page: currentPage,
    per_page: itemsPerPage,
  });

  // Derived data
  const quickCategories = useMemo(() => categories.slice(0, 4), [categories]);
  const categoryLabels = useMemo(() => ['All', ...categories.map(c => c.label)], [categories]);
  
  // Handlers
  const handleSearchChange = useCallback((e: React.ChangeEvent<HTMLInputElement>) => {
    setRawSearch(e.target.value);
    setCurrentPage(1);
  }, []);

  const handleCategoryChange = useCallback((label: string | null) => {
    if (!label) {
      setFilters(prev => ({ ...prev, category_id: null }));
      setCurrentPage(1);
      return;
    }
    const cat = categories.find(c => c.label === label);
    setFilters(prev => ({ ...prev, category_id: cat ? cat.id : null }));
    setCurrentPage(1);
  }, [categories]);

  const handlePriceChange = useCallback((label: string | null) => {
    if (!label) {
      setFilters(prev => ({ ...prev, price_min: null, price_max: null }));
      setCurrentPage(1);
      return;
    }
    const range = PRICE_RANGES.find(r => r.label === label);
    setFilters(prev => ({ 
      ...prev, 
      price_min: range ? range.min : null,
      price_max: range ? range.max : null
    }));
    setCurrentPage(1);
  }, []);

  const handleSortChange = useCallback((val: string) => {
    setFilters(prev => ({ ...prev, sort: val }));
    setCurrentPage(1);
  }, []);

  const handlePageChange = useCallback((page: number) => {
    setCurrentPage(page);
    productGridRef.current?.scrollIntoView({ behavior: 'smooth', block: 'start' });
  }, []);

  const handleProductClick = useCallback((id: string) => {
    checkAuthAndRun(() => navigate(`/product/${encodeURIComponent(id)}`));
  }, [checkAuthAndRun, navigate]);

  const { elementRef: gridRevealRef, isVisible: gridVisible } = useIntersectionReveal({ threshold: 0.1 });

  return (
    <div className="min-h-screen bg-white">
      {/* Hero Section */}
      <section className="px-4 py-4 md:px-10 md:py-8" aria-label="Featured Services">
        <HeroCarousel 
          slides={HERO_SLIDES} 
          onCTA={() => checkAuthAndRun(() => openDiscovery())} 
        />
      </section>

      {/* Categories Section */}
      <section className="px-4 py-12 md:px-20 md:py-24" aria-label="Product Categories">
        <div className="mb-12 flex items-end justify-between">
          <div>
            <h3 className="text-3xl md:text-5xl font-black text-slate-900 uppercase italic tracking-tighter leading-none mb-4 font-display">
              Production Line
            </h3>
            <p className="text-xs md:text-sm font-bold text-slate-400 uppercase tracking-[4px] font-body">
              Select your substrate matrix
            </p>
          </div>
        </div>
        <QuickCategoryRow
          categories={quickCategories}
          isLoading={catLoading}
          onCategoryClick={(id) => checkAuthAndRun(() => openDiscovery(id))}
          onMoreClick={() => checkAuthAndRun(() => openDiscovery())}
        />
      </section>

      {/* Grid Background Texture (CSS Grain) */}
      <div className="relative">
        <div className="absolute inset-0 pointer-events-none opacity-[0.03] fuzzy-overlay" />
        
        {/* Product Marketplace Section */}
        <section 
          ref={productGridRef}
          className="relative px-4 py-20 md:px-20 md:py-32 bg-slate-50/50"
          aria-label="Product Marketplace"
        >
          <div className="mb-20 flex flex-col gap-8">
            <div className="flex flex-col md:flex-row md:items-center justify-between gap-6">
               <h2 className="text-4xl md:text-6xl font-black text-slate-900 uppercase italic tracking-tighter font-display">
                The Marketplace
              </h2>
              <div className="w-full max-w-xl">
                 <SearchBar 
                  value={rawSearch} 
                  onChange={handleSearchChange} 
                  placeholder="Filter catalog by name, SKU, or tag..."
                />
              </div>
            </div>

            {/* Filters Row */}
            <div className="grid grid-cols-2 md:flex md:flex-wrap items-center gap-3">
              <FilterDropdown 
                label="Classification"
                icon={Package}
                options={categoryLabels}
                value={categories.find(c => c.id === filters.category_id)?.label || 'All'}
                onChange={handleCategoryChange}
              />
              <FilterDropdown 
                label="Rate Range"
                icon={Banknote}
                options={PRICE_RANGES.map(r => r.label)}
                value={PRICE_RANGES.find(r => r.min === filters.price_min && r.max === filters.price_max)?.label || 'All Prices'}
                onChange={handlePriceChange}
              />
              <FilterDropdown 
                label="Sort Protocol"
                icon={TrendingUp}
                options={SORT_OPTIONS.map(o => o.label)}
                value={SORT_OPTIONS.find(o => o.value === filters.sort)?.label || 'Newest Arrivals'}
                onChange={(label) => handleSortChange(SORT_OPTIONS.find(o => o.label === label)?.value || 'newest')}
              />
              <FilterDropdown 
                label="Active Status"
                icon={Activity}
                options={STATUS_OPTIONS.map(o => o.label)}
                value={STATUS_OPTIONS.find(o => o.value === filters.status)?.label || 'All Status'}
                onChange={(label) => setFilters(prev => ({ ...prev, status: STATUS_OPTIONS.find(o => o.label === (label as string))?.value || 'all' }))}
              />
              <FilterDropdown 
                label="Owned Plate"
                icon={Printer}
                options={PLATE_OPTIONS.map(o => o.label)}
                value={PLATE_OPTIONS.find(o => o.value === (filters.screenplate_id || 'all'))?.label || 'All Plates'}
                onChange={(label) => setFilters(prev => ({ ...prev, screenplate_id: PLATE_OPTIONS.find(o => o.label === (label as string))?.value || 'all' }))}
              />
            </div>
          </div>

          <div 
            ref={gridRevealRef as React.RefObject<HTMLDivElement>}
            className={`transition-all duration-1000 transform ${gridVisible ? 'translate-y-0 opacity-100' : 'translate-y-12 opacity-0'}`}
          >
            <ProductGrid
              products={products}
              isLoading={prodLoading}
              soldMap={soldMap}
              favorites={[]} // Implement favs if needed, though not required in prompt detail
              onFavClick={() => {}} 
              onProductClick={handleProductClick}
            />

            <Pagination
              currentPage={currentPage}
              totalPages={totalPages}
              onChange={handlePageChange}
            />
          </div>
        </section>
      </div>

      <Footer />
    </div>
  );
};

export default Homepage;
